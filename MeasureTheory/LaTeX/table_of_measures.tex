\RowPad\textbf{Notation} & \textbf{Name} & \textbf{Definition} & \textbf{Integral (or Lower integral)} \\
\hline
\RowPad$m$ & Lebesgue measure & $m (A) = 
      \inf_{\substack{a : \mathbb{N} \to \mathbb{R},\, b : \mathbb{N} \to \mathbb{R},\\ 
        A \subseteq \bigcup_{n} (a_n, b_n)}}
      \sum_{n \in \mathbb{N}} (b_n - a_n)^+$ & 
  $\int_{x \in \mathbb{R}} f(x) \, dx$ \\
\hline 
\RowPad$m|_{(a, b]}$ & Interval measure & $(m|_{(a, b]})(A)=m((a, b]\cap A)$ & $\int_{x \in \mathbb{R}} f(x)\,d(m|_{(a, b]})= \int_{a}^{b} f(x) \, dx$ \\ 
\hline
\RowPad$\mu|_A$ & Restriction & $(\mu|_A)(B)=\mu(A\cap B)$ & $\underline{\int}_{x \in X} f(x)\,d(\mu|_A)
= \underline{\int}_{x \in X} 1_A(x) \cdot f(x)\,d\mu$ \\
\hline
\RowPad$f_*\mu$ & Pushforward & $(f_*\mu)(A)=\mu(f^{-1}(A))$ & $\underline{\int}_{y \in Y} g(y)\,d(f_*\mu)=\underline{\int}_{x \in X} g(f(x))\,d\mu$ \\
\hline
\RowPad$\delta_a$ & Dirac measure & $\delta_a(A)=1_A(a)$ & $\underline{\int}_{x \in X} f(x)\,d\delta_a=f(a)$ \\
\hline
\RowPad$\mu+\nu$ & Addition & $(\mu+\nu)(A)=\mu(A)+\nu(A)$ & $\underline{\int}_{x \in X} f(x)\,d(\mu+\nu)=\underline{\int}_{x \in X} f(x)\,d\mu+\underline{\int}_{x \in X} f(x)\,d\nu$ \\
\hline
\RowPad$\sum_{i\in I}\mu_i$ & Sum & $\Bigl(\sum_{i\in I}\mu_i\Bigr)(A)=\sum_{i\in I}\mu_i(A)$ & $\underline{\int}_{x \in X} f(x)\,d\Bigl(\sum_{i \in I} \mu_i\Bigr)=\sum_{i \in I} \underline{\int}_{x \in X} f(x)\,d\mu_i$ \\
\hline
\RowPad$f\cdot\mu$ & With density $f$ & $(f\cdot\mu)(A)=\underline{\int}_{x \in X} f(x)\,d(\mu|_A)$ & $\underline{\int}_{x \in X} g(x)\,d(f\cdot\mu)=\underline{\int}_{x \in X} f(x) \cdot g(x)\,d\mu$ \\
\hline
\RowPad$\mu \times \nu$ & Product & $(\mu \times \nu) (A) 
      = \underline{\int}_{x \in X} \nu (\{ y \in Y \mid (x,y) \in A \}) \, d\mu$ & $\underline{\int}_{(x,y) \in X \times Y} f (x,y) \, d(\mu \times \nu)
      = \underline{\int}_{x \in X} 
          \underline{\int}_{y \in Y} f (x,y) \, d\nu \, d\mu$ \\
\hline
\RowPad$\mathcal K_\nu(\mu)$ & Kernel integration & $\mathcal K_\nu(\mu)(A)=\underline{\int}_{x \in X} \nu_x(A)\,d\mu$ & $\underline{\int}_{y \in Y} f(y)\,d(\mathcal K_\nu(\mu))=\underline{\int}_{x \in X}\underline{\int}_{y \in Y} f(y)\,d\nu_x\,d\mu$ \\
\hline
